from Bio import SeqIO
from Bio.SeqRecord import SeqRecord

def convert_fasta_to_genbank(fasta_file, genbank_file):
    records = []

    for record in SeqIO.parse(fasta_file, "fasta"):
        genbank_record = SeqRecord(
            record.seq,
            id=record.id,
            name="Example_Gene",
            description=record.description,
            annotations={
                "molecule_type": "DNA",
                "gene": "ExampleGene",
                "function": "Hypothetical protein"
            }
        )

        records.append(genbank_record)

    # Write to GenBank file
    SeqIO.write(records, genbank_file, "genbank")


# Call the function
convert_fasta_to_genbank("example.fasta", "example_output.gb")

# Print file content
with open("example_output.gb", "r") as file:
    print(file.read())


from Bio.Seq import Seq
from Bio.SeqRecord import SeqRecord
from Bio.SeqFeature import SeqFeature, FeatureLocation

# Create SeqRecord
record = SeqRecord(
    Seq("ATGCGTACGTAGCTAGCTAG"),
    id="seq1",
    name="Example_Gene",
    description="An example DNA sequence for gene annotation.",
    annotations={"molecule_type": "DNA"}  # Required
)

# Add annotations
record.annotations["gene"] = "ExampleGene"
record.annotations["function"] = "Hypothetical protein"
record.annotations["organism"] = "Synthetic organism"

# Add feature (gene location)
feature = SeqFeature(
    FeatureLocation(0, 20),  # correct length
    type="gene",
    qualifiers={"gene": "ExampleGene"}
)

record.features.append(feature)

# Print details
print("ID:", record.id)
print("Name:", record.name)
print("Description:", record.description)
print("Annotations:", record.annotations)
print("Features:", record.features)

from Bio import Phylo, AlignIO
from Bio.Phylo.TreeConstruction import DistanceCalculator, DistanceTreeConstructor
import io
 
# Inline clustal alignment (no external file needed)
clustal_data = """\
CLUSTAL W multiple sequence alignment
 
seq1      ATGCGTACGTA
seq2      ATGCGTACGTC
seq3      ATGCGTACGAG
"""
 
# Step 1: Load the sequence alignment
alignment = AlignIO.read(io.StringIO(clustal_data), "clustal")
 
# Step 2: Compute distance matrix (identity model)
calculator = DistanceCalculator('identity')
distance_matrix = calculator.get_distance(alignment)
 
# Step 3: Build the phylogenetic tree using UPGMA
constructor = DistanceTreeConstructor()
tree = constructor.upgma(distance_matrix)
 
# Step 4: Visualize the tree
Phylo.draw_ascii(tree)



from Bio import PDB

# Download the PDB file automatically
pdbl = PDB.PDBList()
pdbl.retrieve_pdb_file("1TUP", pdir="pdb", obsolete=False)

# Step 1: Create a PDB parser
parser = PDB.MMCIFParser(QUIET=True)

# Step 2: Fetch a protein structure from the PDB (using a known PDB ID)
pdb_id = "1TUP"
structure = parser.get_structure("protein", "pdb/1tup.cif")

# Step 3: Visualize and manipulate the structure
model = structure[0]  # Get the first model (if there are multiple)
chain = model['A']    # Select chain 'A'

# Step 4: Print out information about the selected chain
print(f"Chain {chain.id}:")
for residue in chain:
    print(residue)

# Step 5: Save the structure for visualization in an external tool
io = PDB.PDBIO()
io.set_structure(structure)
